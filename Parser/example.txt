INTEGER X    = 3 
INTEGER Y = 4

GOTO START

ETYKIETA: 
MOVJ P1
MOVS P1
MOVMVS P1
HOPEN
HCLOSE
ADDZ P1, 100
	
START:
IF (X > 0) 
X = X - 1
GOTO ETYKIETA
ENDIF

X = Y + X - 3
IF (X > 0) 
GOTO IFISTRUE 
ENDIF
GOTO ETYKIETA_ELSE
IFISTRUE:
MOVJ P1
MOV P2
GOTO KONIECIFA
ETYKIETA_ELSE:
MOVJ P3
MOVS P3
KONIECIFA:
